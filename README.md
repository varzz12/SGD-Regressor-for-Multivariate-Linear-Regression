from sklearn.linear_model import SGDRegressor
import numpy as np
import matplotlib.pyplot as plt

X = np.array([[1,2],[2,1],[3,4],[4,3],[5,5]])

model = SGDRegressor(max_iter=1000, eta0=0.01, learning_rate='constant')
